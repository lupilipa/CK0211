# Faça um programa que preencha um vetor com 10 elementos numéricos inteiros. Calcule e mostre: 
• todos os números pares; 
• a quantidade de números pares; 
• todos os números ímpares; 
• a quantidade de números ímpares.

algoritmo
    declare x[10] numérico
    pares <- 0
    impares <- 0
    para i <- 1 até 10 Faça
    início
        escreva "digite um número inteiro para a", i, "° posição"
        leia x[i]
        se x[i] % 2 == 0 então pares <- pares + 1
        senão impares <- impares + 1
    fim

    escreva "todos os números pares:"
    para i <- 1 até 10 faça 
    início
        se x[i] % 2 == 0 então
            escreva x[i]
    fim
    escreva "qtd de números pares:", pares

    escreva "todos os números impares:"
    para i <- 1 até 10 faça 
    início
        se x[i] % 2 != 0 então
            escreva x[i]
    fim
    escreva "qtd de números impares:", impares
fim_algoritmo

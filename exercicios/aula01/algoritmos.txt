#média aritmética

algoritmo
    declare n1, n2, media: numerico
    escreva('digite a primeira nota:')
    leia(n1)
    escreva('digite a segunda nota:')
    leia(n2)
    media = (n1+n2)/2
    escreva('media:', media)
    se media < 7 entao
        escreva('reprovado')
    senao
        escreva('aprovado')
fim algoritmo



#número par

fim algoritmo



#convertendo temperatura

algoritmo
    declare tempF, tempC: numerico
    escreva('digite a temperatura em Celsius:')
    leia(tempC)
    tempF = (tempC * 1.8) + 32
    escreva('a temperatura dada, em Fahrenheit é:', tempF)
fim algoritmo



#processamento de repetição

algoritmo
    declare n, soma, nro: numerico
    escreva('digite um numero n:')
    leia(n)
    soma = 0
    nro = 1
    enquanto nro <= n
    soma = soma + nro
    nro = nro + 1
    escreva(soma)
fim algoritmo



#média dos números positivos digitados

algoritmo
    declare n, soma, valor: numerico
    n = 0
    soma = 0
    escreva('digite um valor inicial:')
    leia(valor)
    enquanto valor > 0
    soma = soma + valor
    n = n + 1
    leia(valor)
    escreva(soma/n)
fim algoritmo

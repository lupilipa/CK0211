#formatando números em string
para mostrar em uma formatação específica, usa %

%d #inteiros
%f #números reais
%.<numero_digitos>f #números com <numero_digitos> casas decimais
%e #números reais em notação científica
%s #formatação padrão como texto

#exemplos:
r = 6/5
print('%.1f' % r)

saída: 1.2

r = 6/5
print('inteiro = %d e r = %.1f' % (3, r))

saída: inteiro = 3 e r = 1.2

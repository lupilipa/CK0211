def calculo(sal): 
    perc = float(input("Entre com o percentual de aumento: ")) 
    valor = sal * perc / 100 
    return valor 
sal = float(input(“Entre com o salário: “)) 
aum = calculo(sal) 
novo_sal = sal + aum 
print(“Novo salario é: R$%.2f” % novo_sal)

def eh_par(num): 
    resultado = False 
    if num % 2 == 0: 
        resultado = True 
    return resultado 
print(eh_par(2)) 
print(eh_par(3)) 
print(eh_par(8))
